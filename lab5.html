dt[ temp < -90 | temp > 60, temp := NA_real_]

rep_temp <- by_station[!is.na(temp_mean)][which.min(abs(temp_mean - nation_med$temp_med))]

# ---- Q2. Per-state median coordinates ----
state_medians <- dt[!is.na(STATE) & !is.na(LAT) & !is.na(LON),
  .(lat_med = median(LAT), lon_med = median(LON)),
  by = STATE
]
#Computes each state’s median latitude/longitude: state_medians.
#For each state, the median latitude and longitude were computed across all stations, producing one “central” coordinate per state. These median coordinates (shown as blue dots on the map) represent the statistical geographic centers of the states’ weather station distributions. They provide a robust reference point for each state, less affected by outlier stations at the borders.

# ---- Q3. Per-state station closest to state mean location (+ map) ----
state_means <- dt[!is.na(STATE) & !is.na(LAT) & !is.na(LON),
  .(lat_mean = mean(LAT), lon_mean = mean(LON)),
  by = STATE
]
#Computes each state’s mean lat/lon: state_means.
state_station_coords <- dt[!is.na(STATE) & !is.na(LAT) & !is.na(LON),
  .(lat = mean(LAT, na.rm = TRUE), lon = mean(LON, na.rm = TRUE)),
  by = .(STATE, USAF)
]
# For each state, finds the station whose (lat, lon) is closest (Euclidean) to the state mean: closest_by_state.
near_tbl <- merge(state_station_coords, state_means, by = "STATE", all.x = TRUE)
near_tbl[, dist2 := (lat - lat_mean)^2 + (lon - lon_mean)^2] 
closest_by_state <- near_tbl[order(STATE, dist2)][, .SD[1], by = STATE]
#Plots both sets in leaflet: median points (default color) vs. closest-to-mean stations (red).

m <- leaflet() |>
  addProviderTiles("OpenStreetMap") |>
  addCircleMarkers(data = state_medians,
                   lat = ~lat_med, lng = ~lon_med,
                   radius = 5, opacity = 1, fillOpacity = 0.8,
                   label = ~paste0(STATE, " median (", round(lat_med, 3), ", ", round(lon_med, 3), ")")) |>
  addCircleMarkers(data = closest_by_state,
                   lat = ~lat, lng = ~lon,
                   radius = 5, opacity = 1, fillOpacity = 0.8,
                   label = ~paste0(STATE, " closest-to-mean (USAF=", USAF, ")"),
                   color = "red")
m
#For each state, the mean latitude and longitude were calculated, and the weather station closest to this point was identified. On the leaflet map, the blue dots represent the statistical medians of station coordinates (Q2), while the red dots represent the actual stations nearest to the mean coordinates. In most states, the red and blue points overlap or lie very close, indicating that the statistical center aligns well with real stations. However, in some large or sparsely monitored western states, there are noticeable deviations, highlighting uneven station distributions.

# ---- Q4. Temperature bucketing + summaries ----
state_temp <- dt[, .(
  temp_avg = mean(temp, na.rm = TRUE),
  n_records = .N,
  n_na_temp = sum(is.na(temp)),
  n_stations = uniqueN(USAF)
), by = STATE]

state_temp[, temp_bucket := fifelse(temp_avg < 20, "low",
                           fifelse(temp_avg < 25, "mid", "high"))]

# 1) For each state, computes average temperature and assigns a bucket:
#< 20 → "low", 20–<25 → "mid", ≥ 25 → "high".
state_bucket <- state_temp[, .(STATE, temp_bucket)]

# 2) Joins the bucket back to every row in dt (dt2) so the grouping key is row-aligned.
dt2 <- state_bucket[dt, on = "STATE"]

# 3) Summarizes by temp_bucket: number of states, entries, NA entries, stations, and mean temp/wind/pressure → summary_tbl.

summary_tbl <- dt2[!is.na(temp_bucket),
  .(
    n_states      = uniqueN(STATE),
    n_entries     = .N,
    n_na_entries  = sum(is.na(temp) | is.na(wind.sp) | is.na(atm.press)),
    n_stations    = uniqueN(USAF),
    mean_temp     = mean(temp, na.rm = TRUE),
    mean_wind     = mean(wind.sp, na.rm = TRUE),
    mean_pressure = mean(atm.press, na.rm = TRUE)
  ),
  by = temp_bucket
][order(temp_bucket)]
#When grouping states by average temperature, 12 states fell into the high-temperature group (≥25 °C), 25 into the mid group (20–<25 °C), and 11 into the low group (<20 °C). The high group shows the warmest climates with slightly lower pressures, the mid group represents moderate climates with the lowest wind speeds, and the low group represents colder northern states with the strongest winds and highest pressures.
