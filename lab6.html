# ---- 3) Remove stopwords, numbers, and very short tokens ----
data("stop_words")
tokens_clean <- tokens %>%
  anti_join(stop_words, by = "word") %>%         # remove stopwords
  filter(!str_detect(word, "^[0-9]+$")) %>%      # remove pure numbers
  filter(str_count(word) >= 2)                   # remove 1-letter tokens

# Word frequencies after cleaning
word_counts_clean <- tokens_clean %>%
  count(word, sort = TRUE)
print(head(word_counts_clean, 20))
